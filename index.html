Connect Google Sheets

Follow these steps once to start logging all bookings automatically.

  1. 1Go to sheets.google.com and create a new spreadsheet named White Rose Bookings
  2. 2Click Extensions → Apps Script from the top menu
  3. 3Delete any existing code and paste the script shown below, then click Save & Deploy as Web App (set access to "Anyone")
  4. 4Copy the Web App URL and replace https://script.google.com/macros/s/AKfycby8v18YcX9I44O_sTUJSWH0A0sbJl8c7sbRXUH1DEgtnvoahtho-KhcTZH7hl-ycVt9/exec in this HTML file's script section
function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = JSON.parse(e.postData.contents);
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(["Timestamp","Name","Phone","Room","Guests","Check-In","Check-Out","Method"]);
  }
  sheet.appendRow([data.timestamp,data.name,data.phone,data.room,data.guests,data.checkin,data.checkout,data.method]);
  return ContentService.createTextOutput("OK");
}
